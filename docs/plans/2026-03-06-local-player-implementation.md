```typescript
import { create } from 'zustand';
import { Track } from '../types/track';

interface LibraryState {
  tracks: Track[];
  filteredTracks: Track[];
  sourceFilter: 'all' | 'local' | 's3' | 'drive';
  searchQuery: string;

  setTracks: (tracks: Track[]) => void;
  addTrack: (track: Track) => void;
  removeTrack: (trackId: number) => void;
  setSourceFilter: (filter: 'all' | 'local' | 's3' | 'drive') => void;
  setSearchQuery: (query: string) => void;
  filterTracks: () => void;
}

export const useLibraryStore = create<LibraryState>((set, get) => ({
  tracks: [],
  filteredTracks: [],
  sourceFilter: 'all',
  searchQuery: '',

  setTracks: (tracks) => {
    set({ tracks });
    get().filterTracks();
  },

  addTrack: (track) => {
    set((state) => ({ tracks: [...state.tracks, track] }));
    get().filterTracks();
  },

  removeTrack: (trackId) => {
    set((state) => ({ tracks: state.tracks.filter(t => t.id !== trackId) }));
    get().filterTracks();
  },

  setSourceFilter: (filter) => {
    set({ sourceFilter: filter });
    get().filterTracks();
  },

  setSearchQuery: (query) => {
    set({ searchQuery: query });
    get().filterTracks();
  },

  filterTracks: () => {
    const { tracks, sourceFilter, searchQuery } = get();
    let filtered = tracks;

    if (sourceFilter !== 'all') {
      filtered = filtered.filter(t => t.source === sourceFilter);
    }

    if (searchQuery) {
      const query = searchQuery.toLowerCase();
      filtered = filtered.filter(t =>
        t.title.toLowerCase().includes(query) ||
        t.artist?.toLowerCase().includes(query) ||
        t.album?.toLowerCase().includes(query)
      );
    }

    set({ filteredTracks: filtered });
  },
}));
```

```rust
use serde::{Deserialize, Serialize};
use tauri::State;

#[derive(Debug, Serialize, Deserialize)]
pub struct Track {
    pub id: i64,
    pub title: String,
    pub artist: Option<String>,
    pub album: Option<String>,
    pub year: Option<i32>,
    pub genre: Option<String>,
    pub duration: i32,
    pub file_path: String,
    pub source: String,
    pub album_art_url: Option<String>,
    pub created_at: i64,
    pub updated_at: i64,
}

#[tauri::command]
pub async fn get_all_tracks(app: tauri::AppHandle) -> Result<Vec<Track>, String> {
    let db = app.db().await.map_err(|e| e.to_string())?;

    let tracks = db
        .query("SELECT * FROM tracks ORDER BY title")
        .fetch::<Vec<Track>>()
        .await
        .map_err(|e| e.to_string())?;

    Ok(tracks)
}

#[tauri::command]
pub async fn add_track(
    app: tauri::AppHandle,
    title: String,
    artist: Option<String>,
    album: Option<String>,
    year: Option<i32>,
    genre: Option<String>,
    duration: i32,
    file_path: String,
    source: String,
    album_art_url: Option<String>,
) -> Result<i64, String> {
    let db = app.db().await.map_err(|e| e.to_string())?;
    let now = chrono::Utc::now().timestamp();

    let result = db
        .execute(
            "INSERT INTO tracks (title, artist, album, year, genre, duration, file_path, source, album_art_url, created_at, updated_at)
             VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?)",
            &[&title, &artist, &album, &year, &genre, &duration, &file_path, &source, &album_art_url, &now, &now]
        )
        .await
        .map_err(|e| e.to_string())?;

    Ok(result.last_insert_id())
}

#[tauri::command]
pub async fn delete_track(app: tauri::AppHandle, track_id: i64) -> Result<(), String> {
    let db = app.db().await.map_err(|e| e.to_string())?;

    db.execute("DELETE FROM tracks WHERE id = ?", &[&track_id])
        .await
        .map_err(|e| e.to_string())?;

    Ok(())
}
```

```typescript
import { Howl, Howler } from 'howler';
import { Track } from '../types/track';

export class AudioService {
  private currentHowl: Howl | null = null;
  private volume: number = 0.7;
  private onProgressCallback?: (progress: number) => void;
  private onEndCallback?: () => void;
  private progressInterval: number | null = null;

  constructor() {
    Howler.volume(this.volume);
  }

  loadTrack(url: string) {
    if (this.currentHowl) {
      this.currentHowl.unload();
      this.clearProgressInterval();
    }

    this.currentHowl = new Howl({
      src: [url],
      html5: true,
      onload: () => {
        console.log('Track loaded');
      },
      onplay: () => {
        this.startProgressInterval();
      },
      onpause: () => {
        this.clearProgressInterval();
      },
      onstop: () => {
        this.clearProgressInterval();
      },
      onend: () => {
        this.clearProgressInterval();
        if (this.onEndCallback) {
          this.onEndCallback();
        }
      },
      onerror: (id, error) => {
        console.error('Playback error:', error);
      }
    });
  }

  play() {
    if (this.currentHowl) {
      this.currentHowl.play();
    }
  }

  pause() {
    if (this.currentHowl) {
      this.currentHowl.pause();
    }
  }

  stop() {
    if (this.currentHowl) {
      this.currentHowl.stop();
    }
  }

  seek(position: number) {
    if (this.currentHowl) {
      this.currentHowl.seek(position);
    }
  }

  getSeek(): number {
    if (this.currentHowl) {
      return this.currentHowl.seek() as number;
    }
    return 0;
  }

  getDuration(): number {
    if (this.currentHowl) {
      return this.currentHowl.duration();
    }
    return 0;
  }

  setVolume(volume: number) {
    this.volume = Math.max(0, Math.min(1, volume));
    Howler.volume(this.volume);
  }

  getVolume(): number {
    return this.volume;
  }

  isPlaying(): boolean {
    return this.currentHowl?.playing() ?? false;
  }

  onProgress(callback: (progress: number) => void) {
    this.onProgressCallback = callback;
  }

  onEnd(callback: () => void) {
    this.onEndCallback = callback;
  }

  private startProgressInterval() {
    this.clearProgressInterval();
    this.progressInterval = window.setInterval(() => {
      if (this.onProgressCallback && this.currentHowl) {
        const progress = this.getSeek();
        this.onProgressCallback(progress);
      }
    }, 100);
  }

  private clearProgressInterval() {
    if (this.progressInterval) {
      clearInterval(this.progressInterval);
      this.progressInterval = null;
    }
  }

  cleanup() {
    this.stop();
    if (this.currentHowl) {
      this.currentHowl.unload();
    }
    this.clearProgressInterval();
  }
}
```

```rust
use notify::{RecommendedWatcher, RecursiveMode, Watcher, Event, EventKind};
use std::path::Path;
use std::sync::mpsc::channel;
use tauri::{AppHandle, Emitter};

pub fn watch_folder(app: AppHandle, folder_path: String) -> Result<(), String> {
    let (tx, rx) = channel();

    let mut watcher: RecommendedWatcher = Watcher::new(
        move |res: Result<Event, notify::Error>| {
            if let Ok(event) = res {
                let _ = tx.send(event);
            }
        },
        notify::Config::default(),
    ).map_err(|e| e.to_string())?;

    watcher.watch(Path::new(&folder_path), RecursiveMode::Recursive)
        .map_err(|e| e.to_string())?;

    // Spawn thread to handle events
    std::thread::spawn(move || {
        loop {
            match rx.recv() {
                Ok(event) => {
                    match event.kind {
                        EventKind::Create(_) => {
                            for path in event.paths {
                                if let Some(ext) = path.extension() {
                                    if ext == "mp3" {
                                        let path_str = path.to_string_lossy().to_string();
                                        let _ = app.emit("file-created", path_str);
                                    }
                                }
                            }
                        }
                        EventKind::Remove(_) => {
                            for path in event.paths {
                                let path_str = path.to_string_lossy().to_string();
                                let _ = app.emit("file-deleted", path_str);
                            }
                        }
                        _ => {}
                    }
                }
                Err(_) => break,
            }
        }
    });

    Ok(())
}
```

```rust
use std::fs;
use std::path::Path;
use tauri::AppHandle;
use crate::file_watcher;

#[tauri::command]
pub async fn scan_folder(app: AppHandle, folder_path: String) -> Result<Vec<String>, String> {
    let path = Path::new(&folder_path);

    if !path.exists() {
        return Err("Folder does not exist".to_string());
    }

    let mut mp3_files = Vec::new();

    fn visit_dirs(dir: &Path, mp3_files: &mut Vec<String>) -> std::io::Result<()> {
        if dir.is_dir() {
            for entry in fs::read_dir(dir)? {
                let entry = entry?;
                let path = entry.path();
                if path.is_dir() {
                    visit_dirs(&path, mp3_files)?;
                } else if let Some(ext) = path.extension() {
                    if ext == "mp3" {
                        mp3_files.push(path.to_string_lossy().to_string());
                    }
                }
            }
        }
        Ok(())
    }

    visit_dirs(path, &mut mp3_files).map_err(|e| e.to_string())?;

    Ok(mp3_files)
}
```

```typescript
import jsmediatags from 'jsmediatags';

export interface ID3Tags {
  title: string;
  artist: string | null;
  album: string | null;
  year: number | null;
  genre: string | null;
  albumArt: string | null; // base64 data URL
}

export function parseID3Tags(arrayBuffer: ArrayBuffer): Promise<ID3Tags> {
  return new Promise((resolve, reject) => {
    jsmediatags.read(arrayBuffer, {
      onSuccess: (tag) => {
        const tags = tag.tags;

        let albumArt: string | null = null;
        if (tags.picture) {
          const picture = tags.picture;
          const base64 = btoa(
            String.fromCharCode(...new Uint8Array(picture.data))
          );
          albumArt = `data:${picture.format};base64,${base64}`;
        }

        resolve({
          title: tags.title || 'Unknown',
          artist: tags.artist || null,
          album: tags.album || null,
          year: tags.year ? parseInt(tags.year) : null,
          genre: tags.genre || null,
          albumArt,
        });
      },
      onError: (error) => {
        console.error('ID3 parsing error:', error);
        reject(error);
      }
    });
  });
}
```

```rust
use serde::{Deserialize, Serialize};
use tauri::AppHandle;

#[derive(Debug, Serialize, Deserialize)]
pub struct Playlist {
    pub id: i64,
    pub name: String,
    pub created_at: i64,
    pub updated_at: i64,
}

#[tauri::command]
pub async fn get_all_playlists(app: AppHandle) -> Result<Vec<Playlist>, String> {
    let db = app.db().await.map_err(|e| e.to_string())?;

    let playlists = db
        .query("SELECT * FROM playlists ORDER BY name")
        .fetch::<Vec<Playlist>>()
        .await
        .map_err(|e| e.to_string())?;

    Ok(playlists)
}

#[tauri::command]
pub async fn create_playlist(app: AppHandle, name: String) -> Result<i64, String> {
    let db = app.db().await.map_err(|e| e.to_string())?;
    let now = chrono::Utc::now().timestamp();

    let result = db
        .execute(
            "INSERT INTO playlists (name, created_at, updated_at) VALUES (?, ?, ?)",
            &[&name, &now, &now]
        )
        .await
        .map_err(|e| e.to_string())?;

    Ok(result.last_insert_id())
}

#[tauri::command]
pub async fn add_track_to_playlist(
    app: AppHandle,
    playlist_id: i64,
    track_id: i64,
    position: i32,
) -> Result<(), String> {
    let db = app.db().await.map_err(|e| e.to_string())?;
    let now = chrono::Utc::now().timestamp();

    db.execute(
        "INSERT INTO playlist_tracks (playlist_id, track_id, position, created_at) VALUES (?, ?, ?, ?)",
        &[&playlist_id, &track_id, &position, &now]
    )
    .await
    .map_err(|e| e.to_string())?;

    Ok(())
}

#[tauri::command]
pub async fn get_playlist_tracks(app: AppHandle, playlist_id: i64) -> Result<Vec<crate::commands::db::Track>, String> {
    let db = app.db().await.map_err(|e| e.to_string())?;

    let tracks = db
        .query(
            "SELECT t.* FROM tracks t
             JOIN playlist_tracks pt ON t.id = pt.track_id
             WHERE pt.playlist_id = ?
             ORDER BY pt.position"
        )
        .bind(playlist_id)
        .fetch::<Vec<crate::commands::db::Track>>()
        .await
        .map_err(|e| e.to_string())?;

    Ok(tracks)
}

#[tauri::command]
pub async fn delete_playlist(app: AppHandle, playlist_id: i64) -> Result<(), String> {
    let db = app.db().await.map_err(|e| e.to_string())?;

    db.execute("DELETE FROM playlists WHERE id = ?", &[&playlist_id])
        .await
        .map_err(|e| e.to_string())?;

    Ok(())
}
```

```typescript
          <div className="flex justify-between items-center mb-4">
            <h2 className="text-xl font-bold font-mono">{activePlaylist.name}</h2>

              onClick={handlePlayPlaylist}

      )}
    </div>
  );
};
```

```typescript
import React, { useEffect } from 'react';

interface ToastProps {
  message: string;
  type: 'success' | 'error' | 'info';
  onClose: () => void;
  duration?: number;
}

export const Toast: React.FC<ToastProps> = ({ message, type, onClose, duration = 3000 }) => {
  useEffect(() => {
    const timer = setTimeout(onClose, duration);
    return () => clearTimeout(timer);
  }, [duration, onClose]);

  const bgColor = {
    success: 'bg-green-600',
    error: 'bg-red-600',
    info: 'bg-blue-600',
  }[type];

  return (
    <div className={`fixed bottom-4 right-4 ${bgColor} text-white px-6 py-3 rounded-lg shadow-lg z-50 animate-slide-up`}>
      <p>{message}</p>
    </div>
  );
};
```
